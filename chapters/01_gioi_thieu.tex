    \item \textbf{Chương 4:} Thiết kế hệ thống.
    \item \textbf{Chương 5:} Thiết kế cơ sở dữ liệu.
    \item \textbf{Chương 6:} Cài đặt hệ thống.
    \item \textbf{Chương 7:} Kiểm thử hệ thống.
    \item \textbf{Chương 8:} Kết luận và hướng phát triển.
\end{itemize}
