\subsection{Mô tả các lớp chính}
\label{subsec:mo_ta_lop_chinh}

% Sử dụng danh sách liệt kê để làm gọn mục lục
Các lớp dữ liệu chính bao gồm:
\begin{itemize}
    \item \textbf{User}: Lưu trữ thông tin người dùng.
    \item \textbf{Role}: Quản lý vai trò và phân quyền.
    \item \textbf{Series}: Lưu trữ thông tin bộ truyện Manga.
    \item \textbf{Chapter}: Quản lý các chương truyện của bộ truyện.
    \item \textbf{Page}: Quản lý các trang bản thảo vẽ trong một chương.
    \item \textbf{Task}: Thông tin các công việc được giao trong quy trình.
    \item \textbf{Submission}: Quản lý các bài nộp, bản thảo được tải lên.
    \item \textbf{Review}: Đánh giá và nhận xét của Tantou Editor hoặc Editorial Board.
    \item \textbf{Vote}: Ghi nhận lượt bình chọn từ phía người đọc.
    \item \textbf{Ranking}: Quản lý bảng xếp hạng độ phổ biến của các bộ truyện.
    \item \textbf{Notification}: Thông báo hệ thống cho các bên liên quan.
\end{itemize}
