\subsection{Đặc tả Use Case}
\label{subsec:dac_ta_use_case}

% Sử dụng danh sách liệt kê để tránh làm sâu mục lục
Các Use Case cần đặc tả chi tiết:
\begin{enumerate}
    \item \textbf{Đăng nhập hệ thống}
    \item \textbf{Quản lý Series}
    \item \textbf{Quản lý Chapter}
    \item \textbf{Quản lý Task}
    \item \textbf{Nộp Submission}
    \item \textbf{Review Chapter}
    \item \textbf{Xuất bản Manga}
    \item \textbf{Quản lý Ranking}
    \item \textbf{Quản lý Notification}
\end{enumerate}
