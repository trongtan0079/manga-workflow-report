\subsection{ERD (Entity Relationship Diagram)}
\label{subsec:erd}
Sơ đồ mối quan hệ thực thể (ERD) thể hiện cấu trúc cơ sở dữ liệu của hệ thống, mô tả các thực thể chính và mối liên kết giữa chúng.

\subsection{Mô tả các mối quan hệ}
\label{subsec:mo_ta_quan_he}
\begin{itemize}
\item \textbf{User - Role}(1-N): Một người dùng nhiều vai trò, nhưng một vai trò chỉ thuộc về một người dùng
\item \textbf{User - Series}(1-N): Một người dùng có thể tạo nhiều series, nhưng mỗi series chỉ thuộc về một người dùng
\item \textbf{User - Tasks}(1-N): Một người dùng có thể có nhiều task, nhưng mỗi task chỉ thuộc về một người dùng
\item \textbf{Series - Chapter}(1-N): Một series có nhiều chapter, nhưng mỗi chapter chỉ thuộc về một series
\item \textbf{Chapter - Tasks}(1-N): Mõi chapter có nhiều task, nhưng mỗi task chỉ thuộc về một chapter
\item \textbf{User - Submission}(1-N): Một User có thể có nhiều submission, nhưng mỗi submission chỉ thuộc về một user
\end{itemize}

\section{Thiết kế giao diện}
\label{sec:thiet_ke_giao_dien}

% Sử dụng danh sách liệt kê để làm gọn mục lục
Các giao diện chính cần thiết kế (Mockup/Wireframe):
\begin{itemize}
    \item \textbf{Dashboard} (Bảng điều khiển chung cho từng vai trò)
    \item \textbf{Đăng nhập} (Giao diện đăng nhập và phân quyền)
    \item \textbf{Quản lý Series} (Tạo truyện mới, chỉnh sửa thông tin)
    \item \textbf{Quản lý Chapter} (Tạo chương, nộp bản vẽ)
    \item \textbf{Quản lý Task} (Giao việc cho trợ lý, theo dõi tiến độ)
    \item \textbf{Submission} (Đăng tải bản thảo vẽ tay hoặc vẽ số)
    \item \textbf{Review} (Đánh giá, phê duyệt và phản hồi)
    \item \textbf{Notification} (Thông báo trạng thái công việc và bài nộp)
    \item \textbf{Ranking} (Biểu đồ xếp hạng bộ truyện)
\end{itemize}
